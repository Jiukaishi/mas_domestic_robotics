cmake_minimum_required(VERSION 2.8.3)
project(mdr_sciroc_elevator)

find_package(catkin REQUIRED COMPONENTS
  rospy
  roslint
  std_msgs
  geometry_msgs
  mas_execution_manager
  mas_knowledge_base
  mdr_listen_action
  mdr_speech_matching
  message_generation
)

catkin_python_setup()

add_service_files(
    FILES
    RasaInterpreter.srv
)

generate_messages(
    DEPENDENCIES
    std_msgs
)
catkin_package(
  CATKIN_DEPENDS
    rospy
    std_msgs
    geometry_msgs
    mas_execution_manager
    mas_knowledge_base
    mdr_listen_action
    mdr_speech_matching
    message_runtime
)


include_directories()

install(DIRECTORY ros/launch/
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}/ros/launch
)
